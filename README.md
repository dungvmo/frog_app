# Install
```bash
 cd frog/
 -create virtual environment
 python3 -m venv env
 source env/bin/activate
 pip install -r requirements.txt
 
 cd ../frontend/my-app/
 npm install
```

 
 # Run
 ```bash
  from frontend/my-app/ run
  npm start

  from frog_app/frog/ run
  python manage.py runserver
  ```
